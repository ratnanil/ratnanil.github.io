dateTime = {
  let current = new Date();
  let cDate = current.getFullYear() + '-' + (current.getMonth() + 1) + '-' + current.getDate()
  let cTime = current.getHours() + ":" + current.getMinutes() + ":" + current.getSeconds()
  return(cDate + ' ' + cTime)
}

gis2 = gis.items.map(function(x){

  ]
}).flat()

}).flat()
